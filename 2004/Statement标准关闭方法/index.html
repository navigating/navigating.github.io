Statement标准关闭方法

在O/R Mapper的代码中，Statement的关闭模式：

clearup的实现：

/**

 * Close database resource.

 *

 * @param conn

 * @param pstmt

 */

protected void clearup(Connection conn, PreparedStatement pstmt)

{

    if (pstmt != null)

    {

        try

        {

            pstmt.close();

        }

        catch (SQLException se)

        {

            se.printStackTrace();

        }

        pstmt = null;

    }



    if (conn != null)

    {

        try

        {

            conn.close();

        }

        catch (SQLException se)

        {

            se.printStackTrace();

        }

        conn = null;

    }

}

1.一般模式

    Connection conn = null;

PreparedStatement pstmt = null;

try

{

    conn = getConnection();

    pstmt = conn.prepareStatement(SQL);

    pstmt.setString(1, id);

    pstmt.setString(2, type);

    pstmt.executeUpdate();          

} catch (SQLException se)

{

throw new ApplicationException(se);

} finally

{

    clearup(conn, pstmt);//有效的close了Connection;只有一个pstmt,相当于只打开了一个光标，在这里有效的close了pstmt.

}
